Info

Um apontador é uma referência para um valor de um dado tipo.

Um apontador para uma variável contém o endereço que esta variável ocupa em memória.
* -> para ir buscar o valor.

Atenção

*p -> valor
p -> endereço

Declarar um apontador
...
Example
C
int *a, *b; // a e b são apontadores para inteiros.
float *f; // f é apontador para float.
Referenciar uma variável
...
Note

A referência para uma variável é obtida pelo operador &.

Example
C
int i = 100;
int *ip = i;
Utilização do operador *
...
Note

O operador * serve para obter o valor que está a ser apontado por um apontador.

Example
C
int i = 100;
int  *ip = &i;

ou

Example
C
int i = 100;
int  *ip; // definição do apontador.
ip = &i; // inicializações do apontador.
Inicialização de um apontador com outro
...
Note

Um apontador pode ser inicializado com outro do mesmo tipo, neste caso não se usa o operador &.

Example
C
int i = 100;
int *ip = &i;
int *ip2 = ip; //Inicialização de um apontador com outro;
Exemplo da utilização de apontadores
...
Example
C
int i = 100;
int *ip = &i;
int k = *ip; // --> k = 300;
*ip = 300; // --> i = 300;
*ip = *ip+1; // --> i = 301;
Operações ilegais
...
C
&(a+b);
C
int i = 100;
int *ip = i;
Note

Ilegal porque temos que apontar para o endereço da variável, não para a variável;
Teria que ser *ip = &i;

C
int i = 100, *ip = &i;
double *dp = &i;
Note

Ilegal porque são de tipos diferentes, tem que apontar para algo do mesmo tipo.

O que se pode e não se pode fazer
...
C
int *ip;
float *fp;
void *gp;
int i;
float f;
ip = &i;
fp = &f;
fp = &i; // --->> Não se pode igual o float ao int
gp = &i;
gp = &f;
gp = ip;
gp = fp;
i = *gp; // --->> Não se pode igualar a variável ao apontador
f = *gp; // --->> Não se pode igualar a variável ao apontador

Info

Fazer cast de um ponteiro
...
Example
C
int i = 100;
void *gp = &i;
int k;

k = *(int *)gp;

printf("\nK: %d\tGP: %d\n\n", k, *(int *)gp);
Exemplo de mudança de memoria
...
Example
C
#include<stdio.h>

int i = 100;
int j = 200;
int k = 300;

int main(){
    int *ip;
    ip = &i;

    printf("\ni -> Endereco: %d Valor: %d", ip, *ip);
    ip = &j;
    printf("\nj -> Endereco: %d Valor: %d", ip, *ip);
    ip = &k;
    printf("\nk -> Endereco: %d Valor: %d", ip, *ip);
    printf("\n\n");
    ip = &i;
    ip = ip + 2; // Ao somar 2 ao endereço, passa para 2 endereços à frente.
    printf("ip = ip + 2\n");
    printf("ip -> Endereco IP = %d Valor: %d\n\n", ip, *ip);
    *ip = *ip + 2; // *ip + 2 é somar o valor
    printf("*ip = *ip + 2\n");
    printf("ip -> Endereco IP = %d Valor: %d\n", ip, *ip);
    printf("k -> Endereço IP = %d Valor: %d\n\n", ip, k);
}
Passagem de parâmetros de uma função por referência
...
Example
C
#include<stdio.h>

void troca(int *x, int *y){
    int tmp;
    tmp = *x;
    *x = *y;
    *y = tmp;
}

int main(){

    int a=3, b=5;
    printf("\n\n-Antes da funcao-\n");
    printf("a: %d\n", a);
    printf("b: %d\n", b);
    troca(&a,&b);
    printf("\n-Depois da funcao-\n");
    printf("a: %d\n", a);
    printf("b: %d\n", b);
}
Passagem de parâmetros de uma função por valor
...
Warning

Neste exemplo, a troca não funciona

Example
C
void troca(int x, int y){
    int tmp;
    tmp = x;
    x=y;
    y=tmp;
}

int main(){
    int a = 3, b=5;
    troca(a,b);
    printf("a: %d\n", a);
    printf("b: %d\n", b);
}

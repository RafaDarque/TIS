Exemplo -> Programa para pedir um numero inteiro e apresentar o valor dele se este for igual a 10, apresentar o dobro se for igual a 5, e erro se não for nem 10 nem 5.
C
#include<stdio.h>

int main(){
	
	int n;
	
	printf("Introduza um numero inteiro positivo: ");
	scanf("%d", &n);
	
	switch(n){

		case 5:
		printf("\nO dobro de 5 e %d\n", n*2);
		break;
	
		case 10:
		printf("\nO valor introduzido foi %d\n", n);
		break;
	
		default:
		printf("\nNao introduziu valores validos\n");
		break;
	}
	return(0);
}

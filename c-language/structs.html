É um tipo de dados agregados definidos pelo programador, e pode conter diversos elementos de dados de diferentes tipos. Numa estrutura vários tipos de dados coexistem em simultâneo.

C
struct[nome]{
    tipo variável;
    tipo variável;
}[lista de variáveis]; //o que está entre "[]" é opcional

Exemplo 1:

C
struct{
    int dia;
    int mes;
    int ano;
}natal, pascoa;

Exemplo 2:

C
struct data{
    int dia;
    int mes;
    int ano;
};
//Declaração das variáveis
struct data natal, pascoa;
Estruturas dentro de estruturas
...

Exemplo

C
struct data{
    int dia;
    int mes;
    int ano;
}natal, pascoa;

struct aluno{
    int numero;
    char nome[50];
    struct data data_inscricao;
};
struct aluno a, b;

As estruturas não podem conter instâncias de si próprias, mas podem conter apontadores para instâncias de si próprias.

Operações com estruturas:
...

-> Atribuição (globalmente)
-> Obter Endereço
-> Aceder aos elementos das estruturas

Operações não possíveis:
...

-> Diferenciação (== , !=)
-> Comparação (<, >, <=, >=)

Acesso a estruturas
...

O acesso à estrutura faz-se através do operador .

Exemplo 1:

C
struct data{
    int dia;
    int mes;
    int ano;
} natal;

natal.dia = 25;
natal.mes = 12;
natal.ano = 2023;

Exemplo 2:

C
struct data{
     int dia;
     int mes;
     int ano;
}natal, *p = &natal;

(*p).dia = 25;
(*p).mes = 12;
(*p).ano = 2023;

É necessário colocar (*p) porque o . tem prioridade sobre o *

ou

C
struct data{
     int dia;
     int mes;
     int ano;
}natal, *p = &natal;

p->dia = 25;
p->mes = 12;
p->ano = 2023;

Crie uma estrutura para armazenar dados de 20 formandos. Essa estrutura deverá conter os campos número, nome, data de nascimento e a média das disciplinas. As disciplinas devem ser armazenadas numa estrutura onde conste o nome e a nota. NOTA: As disciplinas são CEB, CCD, CFE e CFC. Pedir para preencher os vários campos das estruturas e apresentar o nome e a média dos vários formandos

C
#include<stdio.h>
#include<string.h>

int main(){

    struct disciplina{
        char *nome;
        float nota;
    }ceb, ccd, cfe, cfc;

    struct formando{
        int numero;
        char nome[20];
        int diaNascimento;
        int mesNascimento;
        int anoNascimento;
        float media;
        struct disciplina ufcd[4];
    }formandos[20];

    int i = 0;
    int j = 0;

    for(i=0; i<20; i++){

        formandos[i].ufcd[0].nome = "ceb";
        formandos[i].ufcd[1].nome = "ccd";
        formandos[i].ufcd[2].nome = "cfe";
        formandos[i].ufcd[3].nome = "cfc";
        printf("\n\n--Formando %d--\n\n", i+1);
        printf("Nome: ");
        gets(formandos[i].nome);
        fflush(stdin);
        printf("Data de Nascimento\nDia: ");
        scanf("%d", &formandos[i].diaNascimento);
        printf("Mes: ");
        scanf("%d", &formandos[i].mesNascimento);
        printf("Ano: ");
        scanf("%d", &formandos[i].anoNascimento);
        formandos[i].media = 0;

        for(j=0; j<4; j++){
            printf("Insira nota de '%s': ", formandos[i].ufcd[j].nome);
            scanf("%f", &formandos[i].ufcd[i].nota);
            formandos[i].media = formandos[i].media + formandos[i].ufcd[i].nota;
        }

        formandos[i].media = formandos[i].media / 4;
        fflush(stdin);

    }

    printf("\n\n");
    for(i=0; i<20; i++){
        printf("O aluno %s tem media de: %.2f\n", formandos[i].nome, formandos[i].media);
    }
    printf("\n\n");
}

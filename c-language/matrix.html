Conjunto de células com coordenadas
Usam-se normalmente 2 ciclos for (1 para as linhas e outro para as colunas).

Example
C
//Declarar uma matriz 4x5
//preencher a matriz com números aleatórios (multiplos de 3 | entre 0 e 100)

#include<stdio.h>
#include<time.h>
#include<stdlib.h>

int main(){

	int matriz[4][5]; // 4 linhas e 5 colunas
	int i = 0;
	int j = 0;
	int num = 0;

	srand(time(NULL)); //seed
	printf("\t");

	for(j=0; j<5; j++){
		printf("j(%d) ", j);
	}
	
	printf("\n\n\n");
	
	for(i; i<4; i++){
		printf("i(%d)\t ", i);
		for(j=0; j<5; j++){
			num = 0;
			num = rand()%101;
			while(!(num%3==0)){
				num = rand()%101;
			}
		matriz[i][j] = num;
		printf("%02d | ",num);
	}
	printf("\n\t ------------------------ \n");
	//printf("\n\n");
	}
}
